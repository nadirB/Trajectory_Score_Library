@proc_def ::SpeakersConfig($speakersmap){
	forall $speaker, $aed in $speakersmap {
		$prefix := "/speaker/" + $speaker + "/aed"
		oscsend spat5 "127.0.0.1" : 4072 $prefix
		spat5 $aed
		;print_speakers $speaker $aed
	}
}

@proc_def ::XY_SpeakersState($speakersmap){
	forall $speaker, $xyz in $speakersmap
	{
		$prefix := "/speaker/" + $speaker + "/xyz"
		oscsend spat5 "127.0.0.1" : 4072 $prefix
		spat5 $xyz
		;print_speakers $speaker $aed
	}
}
